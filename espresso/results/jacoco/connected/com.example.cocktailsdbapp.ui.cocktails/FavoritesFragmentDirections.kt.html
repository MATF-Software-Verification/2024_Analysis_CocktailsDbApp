package com.example.cocktailsdbapp.ui.cocktails

import android.os.Bundle
import androidx.navigation.NavDirections
import com.example.cocktailsdbapp.NavigationFavoritesDirections
import com.example.cocktailsdbapp.R
import kotlin.Int
import kotlin.String

public class FavoritesFragmentDirections private constructor() {
  private data class ActionFavoritesFragmentToCategoryFragment(
    public val filter: String = "",
  ) : NavDirections {
    public override val actionId: Int = R.id.action_favoritesFragment_to_CategoryFragment

    public override val arguments: Bundle
      get() {
        val result = Bundle()
        result.putString("filter", this.filter)
        return result
      }
  }

  private data class ActionFavoritesFragmentToCocktailDetailsFragment(
    public val cocktailId: String = "",
  ) : NavDirections {
    public override val actionId: Int = R.id.action_favoritesFragment_to_CocktailDetailsFragment

    public override val arguments: Bundle
      get() {
        val result = Bundle()
        result.putString("cocktailId", this.cocktailId)
        return result
      }
  }

  public companion object {
    public fun actionFavoritesFragmentToCategoryFragment(filter: String = ""): NavDirections =
        ActionFavoritesFragmentToCategoryFragment(filter)

    public fun actionFavoritesFragmentToCocktailDetailsFragment(cocktailId: String = ""):
        NavDirections = ActionFavoritesFragmentToCocktailDetailsFragment(cocktailId)

    public fun actionGlobalCategoryFragment(filter: String = ""): NavDirections =
        NavigationFavoritesDirections.actionGlobalCategoryFragment(filter)

    public fun actionGlobalSearchFragment(): NavDirections =
        NavigationFavoritesDirections.actionGlobalSearchFragment()
  }
}
