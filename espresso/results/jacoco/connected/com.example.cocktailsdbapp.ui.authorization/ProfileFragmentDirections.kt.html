package com.example.cocktailsdbapp.ui.authorization

import android.os.Bundle
import androidx.navigation.NavDirections
import com.example.cocktailsdbapp.NavigationProfileDirections
import com.example.cocktailsdbapp.R
import kotlin.Int
import kotlin.String

public class ProfileFragmentDirections private constructor() {
  private data class ActionProfileFragmentToCategoryFragment(
    public val filter: String = "",
  ) : NavDirections {
    public override val actionId: Int = R.id.action_profileFragment_to_CategoryFragment

    public override val arguments: Bundle
      get() {
        val result = Bundle()
        result.putString("filter", this.filter)
        return result
      }
  }

  public companion object {
    public fun actionProfileFragmentToCategoryFragment(filter: String = ""): NavDirections =
        ActionProfileFragmentToCategoryFragment(filter)

    public fun actionGlobalSearchFragment(): NavDirections =
        NavigationProfileDirections.actionGlobalSearchFragment()

    public fun actionProfileFragmentToInitialState(): NavDirections =
        NavigationProfileDirections.actionProfileFragmentToInitialState()

    public fun actionGlobalCategoryFragment(filter: String = ""): NavDirections =
        NavigationProfileDirections.actionGlobalCategoryFragment(filter)
  }
}
