package com.example.cocktailsdbapp.ui.category

import android.os.Bundle
import androidx.navigation.NavDirections
import com.example.cocktailsdbapp.NavigationCocktailsDirections
import com.example.cocktailsdbapp.R
import kotlin.Int
import kotlin.String

public class FilterFragmentDirections private constructor() {
  private data class ActionFilterFragmentToCocktailsFragment(
    public val filterCategory: String = "Alcoholic or not",
    public val filter: String = "Alcoholic",
  ) : NavDirections {
    public override val actionId: Int = R.id.action_filterFragment_to_CocktailsFragment

    public override val arguments: Bundle
      get() {
        val result = Bundle()
        result.putString("filterCategory", this.filterCategory)
        result.putString("filter", this.filter)
        return result
      }
  }

  public companion object {
    public fun actionFilterFragmentToCocktailsFragment(filterCategory: String = "Alcoholic or not",
        filter: String = "Alcoholic"): NavDirections =
        ActionFilterFragmentToCocktailsFragment(filterCategory, filter)

    public fun actionGlobalCategoryFragment(filter: String = ""): NavDirections =
        NavigationCocktailsDirections.actionGlobalCategoryFragment(filter)

    public fun actionGlobalSearchFragment(): NavDirections =
        NavigationCocktailsDirections.actionGlobalSearchFragment()
  }
}
